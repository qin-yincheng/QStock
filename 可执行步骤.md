## 〇、架构总览：为什么以 czsc 为中心

### 之前方案的问题

之前的方案是"自己写数据获取 + 自己写缠论识别 + 自己写回测"，存在大量重复造轮子：

| 模块 | 之前方案 | czsc 内置能力 |
|------|---------|-------------|
| 数据获取 | 自写 data_fetcher（baostock + akshare） | **内置 ts_connector**，直接对接 Tushare，含60分钟K线、复权处理、数据缓存 |
| 缠论识别 | 自写简化版 ChanAnalyzer | **CZSC 核心对象**，业界最成熟的分型/笔/线段/中枢识别 |
| 信号系统 | 自写 signal_engine | **内置 signals 模块**，数十个信号函数（MACD、均线、笔、分型等） |
| 回测引擎 | 自写 SimpleBacktest | **CzscTrader**，支持多级别联立决策、仓位管理、权重回测 |
| Agent Skill | 从零编写 | **czsc_skills**，现成的缠论思维技能 + 分析脚本 |

### 新方案：czsc 生态 + Tushare + 自定义双龙信号

```
Tushare (你已有token)
    ↓ czsc.connectors.ts_connector
czsc CZSC对象 → 分型/笔/线段/中枢识别
    ↓
czsc.signals 内置信号 + 自定义双龙信号
    ↓
CzscTrader 多级别决策 + 回测
    ↓
czsc_skills → OpenClaw Agent Skill
    ↓
定时推送交易信号
```

### Step 1.4 方案B — baostock + 数据适配层（Tushare分钟数据不可用时）

> 如果 Step 1.2 确认 Tushare 无法获取分钟数据，走此步骤。

**操作** — 创建文件 `core/data_adapter.py`：

```python
"""
数据适配层：将 baostock / akshare 数据转为 czsc RawBar 格式
仅在 Tushare 分钟数据不可用时使用此模块
"""
import baostock as bs
import pandas as pd
from czsc import RawBar, Freq
from datetime import datetime, timedelta
from typing import List


def _baostock_code(code: str) -> str:
    """股票代码转 baostock 格式: 600519 -> sh.600519"""
    code = code.replace('.SH', '').replace('.SZ', '').strip()
    if code.startswith(('sh.', 'sz.')):
        return code
    return f"sh.{code}" if code.startswith(('6', '5')) else f"sz.{code}"


def get_bars_from_baostock(stock_code: str, start_date: str, end_date: str,
                            freq: Freq = Freq.F60) -> List[RawBar]:
    """
    从 baostock 获取K线并转为 czsc RawBar 列表
    
    Args:
        stock_code: 如 "600519.SH" 或 "600519"
        start_date: "2025-09-01"
        end_date: "2026-03-19"
        freq: czsc.Freq 枚举
    """
    freq_map = {Freq.F60: "60", Freq.F30: "30", Freq.F15: "15", Freq.F5: "5"}
    bs_freq = freq_map.get(freq, "60")
    bs_code = _baostock_code(stock_code)
    
    # Tushare 格式的 symbol
    ts_symbol = stock_code if '.' in stock_code else (
        f"{stock_code}.SH" if stock_code.startswith(('6', '5')) else f"{stock_code}.SZ"
    )
    
    lg = bs.login()
    rs = bs.query_history_k_data_plus(
        bs_code,
        "date,time,code,open,high,low,close,volume,amount",
        start_date=start_date, end_date=end_date,
        frequency=bs_freq, adjustflag="2"
    )
    
    data_list = []
    while rs.error_code == '0' and rs.next():
        data_list.append(rs.get_row_data())
    bs.logout()
    
    if not data_list:
        print(f"警告: {stock_code} 未获取到数据")
        return []
    
    df = pd.DataFrame(data_list, columns=rs.fields)
    for col in ['open', 'high', 'low', 'close', 'volume', 'amount']:
        df[col] = pd.to_numeric(df[col], errors='coerce')
    
    # 构造 datetime
    df['datetime'] = pd.to_datetime(df['date'] + ' ' + 
        df['time'].str[:2] + ':' + df['time'].str[2:4] + ':' + df['time'].str[4:6])
    df = df.sort_values('datetime').reset_index(drop=True)
    df = df.dropna(subset=['close'])
    
    # 转为 RawBar 列表
    bars = []
    for i, row in df.iterrows():
        bar = RawBar(
            symbol=ts_symbol,
            dt=pd.Timestamp(row['datetime']),
            id=len(bars),
            freq=freq,
            open=float(row['open']),
            close=float(row['close']),
            high=float(row['high']),
            low=float(row['low']),
            vol=int(row['volume']),
            amount=float(row['amount']),
        )
        bars.append(bar)
    
    print(f"baostock: {ts_symbol} 获取 {len(bars)} 条 {freq.value} RawBar")
    return bars


def get_recent_bars_from_akshare(stock_code: str, freq: Freq = Freq.F60) -> List[RawBar]:
    """
    从 akshare 获取最近5天的分钟K线（盘中实时数据）
    """
    import akshare as ak
    
    symbol = stock_code.replace('.SH', '').replace('.SZ', '').replace('sh.', '').replace('sz.', '')
    ts_symbol = stock_code if '.' in stock_code else (
        f"{stock_code}.SH" if stock_code.startswith(('6', '5')) else f"{stock_code}.SZ"
    )
    
    period_map = {Freq.F60: "60", Freq.F30: "30", Freq.F15: "15", Freq.F5: "5"}
    
    df = ak.stock_zh_a_hist_min_em(
        symbol=symbol,
        period=period_map.get(freq, "60"),
        adjust="qfq"
    )
    
    df = df.rename(columns={
        '时间': 'datetime', '开盘': 'open', '收盘': 'close',
        '最高': 'high', '最低': 'low', '成交量': 'volume', '成交额': 'amount'
    })
    df['datetime'] = pd.to_datetime(df['datetime'])
    
    bars = []
    for i, row in df.iterrows():
        bar = RawBar(
            symbol=ts_symbol,
            dt=pd.Timestamp(row['datetime']),
            id=len(bars),
            freq=freq,
            open=float(row['open']),
            close=float(row['close']),
            high=float(row['high']),
            low=float(row['low']),
            vol=int(row['volume']),
            amount=float(row['amount']),
        )
        bars.append(bar)
    
    print(f"akshare: {ts_symbol} 获取 {len(bars)} 条实时 {freq.value} RawBar")
    return bars


def get_merged_bars(stock_code: str, freq: Freq = Freq.F60,
                     history_months: int = 6) -> List[RawBar]:
    """
    实战用：baostock历史 + akshare实时，拼接后返回统一的 RawBar 列表
    """
    end_date = datetime.now().strftime("%Y-%m-%d")
    start_date = (datetime.now() - timedelta(days=history_months * 30)).strftime("%Y-%m-%d")
    
    hist_bars = get_bars_from_baostock(stock_code, start_date, end_date, freq)
    
    try:
        recent_bars = get_recent_bars_from_akshare(stock_code, freq)
    except Exception as e:
        print(f"akshare 实时数据获取失败: {e}")
        return hist_bars
    
    # 合并去重（以 dt 为主键，实时数据优先）
    bar_map = {bar.dt: bar for bar in hist_bars}
    for bar in recent_bars:
        bar_map[bar.dt] = bar
    
    merged = sorted(bar_map.values(), key=lambda b: b.dt)
    # 重新编号
    for i, bar in enumerate(merged):
        bar.id = i
    
    print(f"合并结果: {len(merged)} 条 RawBar")
    return merged
```

> **说明**：此模块仅在 Tushare 无法获取分钟数据时使用。如果 Tushare 方案A可行，则整个项目都不需要 baostock 和 akshare，全部走 czsc 内置 ts_connector。

**产出物**：`core/data_adapter.py`（方案B专用）。

### Step 1.5 统一数据接口

**目标**：封装一个统一接口，根据方案A/B自动选择数据源。

**操作** — 创建文件 `core/data_provider.py`：

```python
"""
统一数据提供者：自动选择最佳数据源
- 方案A (Tushare分钟可用): czsc ts_connector
- 方案B (Tushare分钟不可用): baostock + akshare 适配层
"""
from czsc import RawBar, Freq
from typing import List

# ===== 配置区：根据 Step 1.2 的验证结果设置 =====
TUSHARE_MINUTE_OK = True  # ← 根据验证结果修改：True=方案A, False=方案B
# ================================================


def get_raw_bars(symbol: str, freq: Freq, sdt: str, edt: str,
                  fq: str = "前复权") -> List[RawBar]:
    """
    统一获取K线数据
    
    Args:
        symbol: 方案A用 "600519.SH#E" 格式; 方案B用 "600519.SH" 格式
        freq: czsc.Freq 枚举，如 Freq.F60, Freq.D
        sdt: 开始日期 "20250901"
        edt: 结束日期 "20260319"
        fq: "前复权" 或 "后复权"
    
    Returns:
        List[RawBar]
    """
    if TUSHARE_MINUTE_OK:
        from czsc.connectors.ts_connector import get_raw_bars as ts_get_bars
        # 确保 symbol 含 #asset 后缀
        if '#' not in symbol:
            asset = "I" if symbol.startswith(('000', '399')) and symbol.endswith(('.SH', '.SZ')) else "E"
            symbol = f"{symbol}#{asset}"
        return ts_get_bars(symbol=symbol, freq=freq, sdt=sdt, edt=edt, fq=fq, raw_bar=True)
    else:
        from core.data_adapter import get_bars_from_baostock
        # baostock 需要 "YYYY-MM-DD" 格式
        sdt_fmt = f"{sdt[:4]}-{sdt[4:6]}-{sdt[6:8]}" if len(sdt) == 8 else sdt
        edt_fmt = f"{edt[:4]}-{edt[4:6]}-{edt[6:8]}" if len(edt) == 8 else edt
        code = symbol.split('#')[0] if '#' in symbol else symbol
        return get_bars_from_baostock(code, sdt_fmt, edt_fmt, freq)


def get_realtime_bars(symbol: str, freq: Freq = Freq.F60,
                       history_months: int = 6) -> List[RawBar]:
    """
    实战盘中用：获取历史+实时拼接的完整数据
    
    方案A: czsc ts_connector 直接获取（Tushare 盘中数据有一定延迟但可用）
    方案B: baostock历史 + akshare实时 拼接
    """
    from datetime import datetime, timedelta
    
    edt = datetime.now().strftime("%Y%m%d")
    sdt = (datetime.now() - timedelta(days=history_months * 30)).strftime("%Y%m%d")
    
    if TUSHARE_MINUTE_OK:
        return get_raw_bars(symbol, freq, sdt, edt)
    else:
        from core.data_adapter import get_merged_bars
        code = symbol.split('#')[0] if '#' in symbol else symbol
        return get_merged_bars(code, freq, history_months)
```

**产出物**：`core/data_provider.py` — 方案A/B统一入口。

### Step 3.1 基于 czsc 实现缠论买卖点判断

**目标**：利用 czsc 的笔和分型数据，实现缠论三类买卖点判断。

**操作** — 创建文件 `strategy/chan_signals.py`：

```python
"""
基于 czsc 的缠论买卖点信号模块
利用 CZSC 对象识别的笔和分型，判断三类买卖点
"""
from czsc import CZSC, RawBar
from typing import List, Optional


class ChanSignalGenerator:
    """缠论买卖点信号生成器"""
    
    def __init__(self, bars: List[RawBar]):
        self.czsc = CZSC(bars)
    
    def update(self, bar: RawBar):
        """逐K线更新（实盘用）"""
        self.czsc.update(bar)
    
    def analyze(self) -> dict:
        """生成当前的缠论信号"""
        c = self.czsc
        result = {
            'signal': 'HOLD',
            'detail': '无明确信号',
            'fx_count': len(c.fx_list),
            'bi_count': len(c.bi_list),
            'last_bi_direction': None,
        }
        
        if len(c.bi_list) < 5:
            result['detail'] = f'笔数量不足({len(c.bi_list)}笔)，等待更多数据'
            return result
        
        last_bi = c.bi_list[-1]
        result['last_bi_direction'] = last_bi.direction.value
        
        # 中枢识别：取最近几笔的重叠区间
        zhongshu = self._find_zhongshu(c.bi_list[-7:])
        
        if zhongshu:
            signal, detail = self._check_third_buy_sell(c.bi_list, zhongshu)
            if signal != 'HOLD':
                result['signal'] = signal
                result['detail'] = detail
                return result
        
        # 二买/二卖判断
        signal, detail = self._check_second_buy_sell(c.bi_list)
        if signal != 'HOLD':
            result['signal'] = signal
            result['detail'] = detail
            return result
        
        # 背驰判断（辅助一买/一卖）
        signal, detail = self._check_divergence(c.bi_list)
        if signal != 'HOLD':
            result['signal'] = signal
            result['detail'] = detail
        
        return result
    
    def _find_zhongshu(self, bis) -> Optional[tuple]:
        """识别中枢：3笔以上的重叠价格区间，返回 (ZG, ZD) 或 None"""
        if len(bis) < 3:
            return None
        
        for i in range(len(bis) - 2):
            three_bis = bis[i:i+3]
            highs = [bi.high for bi in three_bis]
            lows = [bi.low for bi in three_bis]
            zg = min(highs)  # 最高点中的最低
            zd = max(lows)    # 最低点中的最高
            if zg > zd:
                return (zg, zd)
        return None
    
    def _check_third_buy_sell(self, bi_list, zhongshu) -> tuple:
        """三买/三卖：离开中枢后回试不破"""
        zg, zd = zhongshu
        if len(bi_list) < 3:
            return ('HOLD', '')
        
        last_bi = bi_list[-1]
        prev_bi = bi_list[-2]
        
        # 三买：向上离开中枢，回踩（最近下降笔）不破 ZG
        if (last_bi.direction.value == 'down' and 
            last_bi.low > zg and
            prev_bi.direction.value == 'up' and 
            prev_bi.high > zg):
            return ('BUY_3', f'三类买点: 回踩不破中枢上沿 ZG={zg:.2f}, 低点={last_bi.low:.2f}')
        
        # 三卖：向下离开中枢，反弹（最近上升笔）不破 ZD
        if (last_bi.direction.value == 'up' and 
            last_bi.high < zd and
            prev_bi.direction.value == 'down' and 
            prev_bi.low < zd):
            return ('SELL_3', f'三类卖点: 反弹不破中枢下沿 ZD={zd:.2f}, 高点={last_bi.high:.2f}')
        
        return ('HOLD', '')
    
    def _check_second_buy_sell(self, bi_list) -> tuple:
        """二买/二卖：一买/一卖后的第一次回调/反弹结束"""
        if len(bi_list) < 4:
            return ('HOLD', '')
        
        last = bi_list[-1]
        prev = bi_list[-2]
        prev2 = bi_list[-3]
        prev3 = bi_list[-4]
        
        # 二买：下降趋势→底部反转→回调不创新低
        if (last.direction.value == 'up' and
            prev.direction.value == 'down' and
            prev.low > prev3.low and  # 回调不创新低
            prev2.direction.value == 'up'):
            return ('BUY_2', f'二类买点: 回调低点{prev.low:.2f} > 前低{prev3.low:.2f}')
        
        # 二卖：上升趋势→顶部反转→反弹不创新高
        if (last.direction.value == 'down' and
            prev.direction.value == 'up' and
            prev.high < prev3.high and
            prev2.direction.value == 'down'):
            return ('SELL_2', f'二类卖点: 反弹高点{prev.high:.2f} < 前高{prev3.high:.2f}')
        
        return ('HOLD', '')
    
    def _check_divergence(self, bi_list) -> tuple:
        """背驰判断：用笔的幅度对比，辅助一买/一卖判断"""
        if len(bi_list) < 4:
            return ('HOLD', '')
        
        # 找最近两笔同向笔
        last = bi_list[-1]
        same_dir = [bi for bi in bi_list[-5:] if bi.direction == last.direction]
        
        if len(same_dir) < 2:
            return ('HOLD', '')
        
        bi_new = same_dir[-1]
        bi_old = same_dir[-2]
        
        amp_new = abs(bi_new.fx_b.fx - bi_new.fx_a.fx)
        amp_old = abs(bi_old.fx_b.fx - bi_old.fx_a.fx)
        
        if amp_old == 0:
            return ('HOLD', '')
        
        # 上涨背驰：价格创新高但幅度减小
        if (last.direction.value == 'up' and 
            bi_new.high > bi_old.high and 
            amp_new < amp_old * 0.8):
            return ('SELL_DIV', f'上涨背驰: 新笔幅度{amp_new:.2f} < 前笔{amp_old:.2f}×0.8')
        
        # 下跌背驰：价格创新低但幅度减小
        if (last.direction.value == 'down' and 
            bi_new.low < bi_old.low and 
            amp_new < amp_old * 0.8):
            return ('BUY_DIV', f'下跌背驰: 新笔幅度{amp_new:.2f} < 前笔{amp_old:.2f}×0.8')
        
        return ('HOLD', '')
```

### Step 3.2 双龙战法信号模块

**目标**：实现 MA金叉 + MACD金叉 的双龙信号。

**操作** — 创建文件 `strategy/dragon_signals.py`：

```python
"""
双龙战法信号模块
第一龙：EMA5 上穿 EMA10（均线金叉）
第二龙：MACD DIF 上穿 DEA（MACD金叉）
"""
import pandas as pd
import pandas_ta as ta
from czsc import RawBar
from typing import List


class DragonSignalGenerator:
    """双龙战法信号生成器"""
    
    def __init__(self, bars: List[RawBar], lookback_window: int = 3):
        self.lookback = lookback_window
        self.df = self._bars_to_df(bars)
        self._calculate()
    
    def _bars_to_df(self, bars: List[RawBar]) -> pd.DataFrame:
        """RawBar 列表转 DataFrame"""
        records = [{
            'datetime': bar.dt, 'open': bar.open, 'high': bar.high,
            'low': bar.low, 'close': bar.close, 'volume': bar.vol
        } for bar in bars]
        return pd.DataFrame(records)
    
    def _calculate(self):
        """计算技术指标"""
        df = self.df
        
        df['ema5'] = ta.ema(df['close'], length=5)
        df['ema10'] = ta.ema(df['close'], length=10)
        
        macd_result = ta.macd(df['close'], fast=12, slow=26, signal=9)
        df['dif'] = macd_result.iloc[:, 0]
        df['dea'] = macd_result.iloc[:, 1]
        df['macd_hist'] = macd_result.iloc[:, 2]
        
        # 金叉/死叉判断
        df['ema_cross_up'] = (df['ema5'] > df['ema10']) & (df['ema5'].shift(1) <= df['ema10'].shift(1))
        df['ema_cross_down'] = (df['ema5'] < df['ema10']) & (df['ema5'].shift(1) >= df['ema10'].shift(1))
        df['macd_cross_up'] = (df['dif'] > df['dea']) & (df['dif'].shift(1) <= df['dea'].shift(1))
        df['macd_cross_down'] = (df['dif'] < df['dea']) & (df['dif'].shift(1) >= df['dea'].shift(1))
        
        # 窗口内信号
        df['ema_buy_window'] = df['ema_cross_up'].rolling(self.lookback).max().fillna(0).astype(bool)
        df['macd_buy_window'] = df['macd_cross_up'].rolling(self.lookback).max().fillna(0).astype(bool)
        
        self.df = df
    
    def get_signal(self) -> dict:
        """获取最新的双龙信号"""
        df = self.df
        last = df.iloc[-1]
        
        signal = 'HOLD'
        detail = ''
        
        # 卖出优先
        if last['ema_cross_down'] or last['macd_cross_down']:
            signal = 'SELL'
            detail = f"死叉: EMA{'↓' if last['ema_cross_down'] else ''} MACD{'↓' if last['macd_cross_down'] else ''}"
        # 强买：双龙齐飞 + MACD柱正 + DEA > 0
        elif last['ema_buy_window'] and last['macd_buy_window'] and last['macd_hist'] > 0 and last['dea'] > 0:
            signal = 'STRONG_BUY'
            detail = f"双龙齐飞(强): EMA5={last['ema5']:.2f}>EMA10={last['ema10']:.2f}, DIF>DEA>0"
        # 普通买：双龙齐飞 + MACD柱正
        elif last['ema_buy_window'] and last['macd_buy_window'] and last['macd_hist'] > 0:
            signal = 'BUY'
            detail = f"双龙齐飞: EMA5={last['ema5']:.2f}>EMA10={last['ema10']:.2f}, MACD柱正"
        
        return {
            'signal': signal,
            'detail': detail,
            'ema5': round(last['ema5'], 2) if pd.notna(last['ema5']) else None,
            'ema10': round(last['ema10'], 2) if pd.notna(last['ema10']) else None,
            'dif': round(last['dif'], 4) if pd.notna(last['dif']) else None,
            'dea': round(last['dea'], 4) if pd.notna(last['dea']) else None,
            'macd_hist': round(last['macd_hist'], 4) if pd.notna(last['macd_hist']) else None,
        }
    
    def get_all_signals(self) -> pd.DataFrame:
        """返回带信号标注的完整DataFrame（用于回测）"""
        df = self.df.copy()
        
        df['dragon_signal'] = 'HOLD'
        
        sell = df['ema_cross_down'] | df['macd_cross_down']
        strong_buy = (df['ema_buy_window'] & df['macd_buy_window'] & 
                      (df['macd_hist'] > 0) & (df['dea'] > 0))
        normal_buy = (df['ema_buy_window'] & df['macd_buy_window'] & 
                      (df['macd_hist'] > 0) & ~strong_buy)
        
        df.loc[sell, 'dragon_signal'] = 'SELL'
        df.loc[strong_buy, 'dragon_signal'] = 'STRONG_BUY'
        df.loc[normal_buy, 'dragon_signal'] = 'BUY'
        
        return df
```

### Step 3.3 信号融合引擎

**目标**：将缠论信号（方向判断）和双龙信号（时机确认）融合。

**操作** — 创建文件 `strategy/signal_engine.py`：

```python
"""
信号融合引擎
缠论判断大方向（趋势/买卖点级别） + 双龙判断入场时机 = 综合信号
"""
from czsc import RawBar, Freq
from typing import List
from strategy.chan_signals import ChanSignalGenerator
from strategy.dragon_signals import DragonSignalGenerator


class SignalEngine:
    """综合信号引擎"""
    
    def __init__(self, bars: List[RawBar]):
        self.bars = bars
        self.chan = ChanSignalGenerator(bars)
        self.dragon = DragonSignalGenerator(bars)
    
    def generate(self) -> dict:
        """生成综合交易信号"""
        chan_result = self.chan.analyze()
        dragon_result = self.dragon.get_signal()
        
        chan_sig = chan_result['signal']
        dragon_sig = dragon_result['signal']
        
        final = self._fuse(chan_sig, dragon_sig)
        
        return {
            'final_signal': final['action'],
            'confidence': final['confidence'],
            'reason': final['reason'],
            'chan_signal': chan_sig,
            'chan_detail': chan_result['detail'],
            'dragon_signal': dragon_sig,
            'dragon_detail': dragon_result['detail'],
            'bi_count': chan_result['bi_count'],
            'last_bi_direction': chan_result['last_bi_direction'],
            'ema5': dragon_result['ema5'],
            'ema10': dragon_result['ema10'],
            'dif': dragon_result['dif'],
            'dea': dragon_result['dea'],
        }
    
    def _fuse(self, chan: str, dragon: str) -> dict:
        """
        融合规则：
        1. 缠论买点 + 双龙强买 → 强烈买入 (90%)
        2. 缠论买点 + 双龙买入 → 买入 (75%)
        3. 双龙强买(无缠论) → 观察 (60%)
        4. 缠论买点(无双龙) → 等待确认 (50%)
        5. 任一卖出 → 卖出 (80%)
        6. 其他 → 持仓等待
        """
        chan_buy = chan.startswith('BUY')
        dragon_buy = dragon in ('BUY', 'STRONG_BUY')
        dragon_strong = dragon == 'STRONG_BUY'
        any_sell = chan.startswith('SELL') or dragon == 'SELL'
        
        if any_sell:
            return {'action': 'SELL', 'confidence': 0.80,
                    'reason': f'卖出信号 [缠论:{chan}, 双龙:{dragon}]'}
        
        if chan_buy and dragon_strong:
            return {'action': 'STRONG_BUY', 'confidence': 0.90,
                    'reason': f'强买: 缠论{chan}+双龙齐飞'}
        
        if chan_buy and dragon_buy:
            return {'action': 'BUY', 'confidence': 0.75,
                    'reason': f'买入: 缠论{chan}+双龙确认'}
        
        if dragon_strong:
            return {'action': 'BUY_WATCH', 'confidence': 0.60,
                    'reason': f'观察: 双龙齐飞但缠论未确认({chan})'}
        
        if chan_buy:
            return {'action': 'WAIT_CONFIRM', 'confidence': 0.50,
                    'reason': f'等待: 缠论{chan}但双龙未共振'}
        
        return {'action': 'HOLD', 'confidence': 0.0,
                'reason': f'无信号 [缠论:{chan}, 双龙:{dragon}]'}


def analyze_stock(symbol: str, sdt: str, edt: str) -> dict:
    """分析单只股票"""
    from core.data_provider import get_raw_bars
    from czsc import Freq
    
    bars = get_raw_bars(symbol, Freq.F60, sdt, edt)
    if not bars or len(bars) < 30:
        return {'error': f'{symbol} 数据不足'}
    
    engine = SignalEngine(bars)
    result = engine.generate()
    result['symbol'] = symbol
    result['datetime'] = str(bars[-1].dt)
    result['close'] = bars[-1].close
    result['bar_count'] = len(bars)
    return result


def analyze_stock_realtime(symbol: str) -> dict:
    """实战盘中分析（自动拼接历史+实时数据）"""
    from core.data_provider import get_realtime_bars
    from czsc import Freq
    
    bars = get_realtime_bars(symbol, Freq.F60)
    if not bars or len(bars) < 30:
        return {'error': f'{symbol} 数据不足'}
    
    engine = SignalEngine(bars)
    result = engine.generate()
    result['symbol'] = symbol
    result['datetime'] = str(bars[-1].dt)
    result['close'] = bars[-1].close
    result['bar_count'] = len(bars)
    return result
```

### Step 4.1 回测引擎（含 T+1 约束 + 印花税）

**操作** — 创建文件 `backtest/backtester.py`：

```python
"""
回测引擎
- T+1 约束：买入当天不能卖出
- 手续费：万三（双向）
- 印花税：千一（仅卖出）
- 滑点：0.1%
"""
import pandas as pd
import numpy as np
from czsc import RawBar, Freq
from typing import List
from strategy.dragon_signals import DragonSignalGenerator


class Backtester:
    """含 T+1 约束的回测引擎"""
    
    def __init__(self, initial_cash=100000, commission=0.0003,
                 stamp_tax=0.001, slippage=0.001):
        self.initial_cash = initial_cash
        self.commission = commission
        self.stamp_tax = stamp_tax
        self.slippage = slippage
    
    def run(self, bars: List[RawBar]) -> dict:
        """运行回测"""
        if len(bars) < 30:
            return {'error': '数据不足'}
        
        dragon = DragonSignalGenerator(bars)
        signals_df = dragon.get_all_signals()
        
        cash = self.initial_cash
        position = 0
        buy_price = 0.0
        buy_date = None
        trades = []
        equity_curve = []
        
        for _, row in signals_df.iterrows():
            price = row['close']
            signal = row['dragon_signal']
            current_date = pd.Timestamp(row['datetime']).date()
            
            equity = cash + position * price
            equity_curve.append({
                'datetime': row['datetime'],
                'equity': equity,
                'price': price,
                'position': position
            })
            
            # 买入
            if signal in ('BUY', 'STRONG_BUY') and position == 0:
                buy_price = price * (1 + self.slippage)
                shares = int(cash / (buy_price * (1 + self.commission)) / 100) * 100
                if shares >= 100:
                    cost = shares * buy_price * (1 + self.commission)
                    cash -= cost
                    position = shares
                    buy_date = current_date
                    trades.append({
                        'datetime': row['datetime'], 'action': 'BUY',
                        'price': round(buy_price, 2), 'shares': shares,
                        'cost': round(cost, 2), 'signal': signal
                    })
            
            # 卖出（T+1 约束：买入当天不能卖出）
            elif signal == 'SELL' and position > 0:
                if buy_date is not None and current_date <= buy_date:
                    continue
                sell_price = price * (1 - self.slippage)
                revenue = position * sell_price * (1 - self.commission - self.stamp_tax)
                profit = revenue - position * buy_price
                cash += revenue
                trades.append({
                    'datetime': row['datetime'], 'action': 'SELL',
                    'price': round(sell_price, 2), 'shares': position,
                    'revenue': round(revenue, 2), 'profit': round(profit, 2),
                    'signal': signal
                })
                position = 0
                buy_date = None
        
        return self._metrics(equity_curve, trades)
    
    def _metrics(self, equity_curve, trades) -> dict:
        eq_df = pd.DataFrame(equity_curve)
        if eq_df.empty:
            return {'error': '无权益数据'}
        
        final = eq_df['equity'].iloc[-1]
        total_return = (final - self.initial_cash) / self.initial_cash
        
        eq_df['peak'] = eq_df['equity'].cummax()
        eq_df['drawdown'] = (eq_df['equity'] - eq_df['peak']) / eq_df['peak']
        max_dd = eq_df['drawdown'].min()
        
        sell_trades = [t for t in trades if t['action'] == 'SELL']
        wins = [t for t in sell_trades if t.get('profit', 0) > 0]
        losses = [t for t in sell_trades if t.get('profit', 0) <= 0]
        
        n_bars = len(eq_df)
        n_years = n_bars / (250 * 4) if n_bars > 0 else 1
        annual_ret = (1 + total_return) ** (1 / n_years) - 1 if n_years > 0 else 0
        win_rate = len(wins) / len(sell_trades) if sell_trades else 0
        
        avg_win = np.mean([t['profit'] for t in wins]) if wins else 0
        avg_loss = abs(np.mean([t['profit'] for t in losses])) if losses else 1
        profit_ratio = avg_win / avg_loss if avg_loss > 0 else float('inf')
        
        return {
            'initial_cash': self.initial_cash,
            'final_equity': round(final, 2),
            'total_return': f"{total_return:.2%}",
            'annual_return': f"{annual_ret:.2%}",
            'max_drawdown': f"{max_dd:.2%}",
            'total_trades': len(sell_trades),
            'win_rate': f"{win_rate:.2%}",
            'profit_ratio': f"{profit_ratio:.2f}",
            'trades': trades,
            'equity_curve': eq_df,
        }


if __name__ == "__main__":
    from core.data_provider import get_raw_bars
    from czsc import Freq
    
    bt = Backtester(initial_cash=100000)
    
    test_stocks = [
        ("600519.SH#E", "贵州茅台"),
        ("000001.SZ#E", "平安银行"),
        ("300750.SZ#E", "宁德时代"),
    ]
    
    print("=" * 60)
    print("    双龙战法 60分钟 回测报告 (含T+1约束)")
    print("=" * 60)
    
    for symbol, name in test_stocks:
        bars = get_raw_bars(symbol, Freq.F60, "20250101", "20260319")
        if not bars:
            print(f"\n{name}: 数据获取失败")
            continue
        
        result = bt.run(bars)
        if 'error' in result:
            print(f"\n{name}: {result['error']}")
            continue
        
        print(f"\n--- {name} ({symbol.split('#')[0]}) ---")
        print(f"  收益: {result['total_return']}  年化: {result['annual_return']}")
        print(f"  回撤: {result['max_drawdown']}  胜率: {result['win_rate']}")
        print(f"  交易: {result['total_trades']}次  盈亏比: {result['profit_ratio']}")
```

**产出物**：`backtest/backtester.py` — 含 T+1 约束的回测引擎。

```python
"""市场扫描：对股票池运行策略分析"""
from strategy.signal_engine import analyze_stock
from datetime import datetime, timedelta

DEFAULT_POOL = [
    "600519.SH#E",  # 贵州茅台
    "000858.SZ#E",  # 五粮液
    "300750.SZ#E",  # 宁德时代
    "601318.SH#E",  # 中国平安
    "600036.SH#E",  # 招商银行
    "002594.SZ#E",  # 比亚迪
    "300059.SZ#E",  # 东方财富
    "600900.SH#E",  # 长江电力
]

def scan(pool=None):
    pool = pool or DEFAULT_POOL
    edt = datetime.now().strftime("%Y%m%d")
    sdt = (datetime.now() - timedelta(days=180)).strftime("%Y%m%d")
    
    results = []
    for symbol in pool:
        try:
            r = analyze_stock(symbol, sdt, edt)
            results.append(r)
            sig = r.get('final_signal', 'ERROR')
            print(f"  {symbol}: {sig} ({r.get('confidence', 0):.0%}) - {r.get('reason', '')}")
        except Exception as e:
            print(f"  {symbol}: 失败 - {e}")
    
    actionable = [r for r in results if r.get('final_signal') not in ('HOLD', 'ERROR', None)]
    print(f"\n扫描完成: {len(pool)}只, 有信号: {len(actionable)}只")
    for r in actionable:
        print(f"  ⚡ {r['symbol']}: {r['final_signal']} - {r['reason']}")
    
    return results

if __name__ == "__main__":
    scan()
```

> **股票池说明**：默认池是硬编码的8只龙头股。建议跟导师确认：
> - 导师是否有自己关注的股票列表？
> - 是否需要全市场扫描（czsc ts_connector 的 `get_symbols()` 可获取全市场股票列表）？

## 避坑指南

### 安装相关
1. **czsc 依赖多**：czsc 0.10.x 有大量依赖（clickhouse-connect, redis 等），部分可能安装失败但不影响核心功能。如果 `pip install czsc` 报错，尝试 `pip install czsc --no-deps` 后手动安装核心依赖
2. **ta-lib 安装**：Windows 下 ta-lib C库安装麻烦，可从 https://github.com/cgohlke/talib-build 下载 `.whl`。双龙战法已用 pandas_ta 替代实现
3. **Python 版本**：czsc 要求 >= 3.10，确认版本后再开始

### 数据相关
4. **Tushare 积分**：分钟数据需要积分，提前在 Tushare 社区刷积分
5. **Tushare 频率限制**：每分钟请求次数有限，数据获取代码中加 sleep
6. **baostock 延迟**：当天数据要到晚上20:30才入库，盘中只能用 akshare

### 策略相关
7. **T+1 必须在回测中实现**：否则60分钟策略回测结果严重失真
8. **印花税**：卖出时千一，不要遗漏
9. **避免过拟合**：不对单只股票反复调参，保持参数通用性

### Agent相关
10. **免费额度管理**：策略开发在本地，Agent集成才用云平台
11. **czsc_skills 是参考**：不能直接在 OpenClaw 中使用（它是为 Claude 设计的），需要适配为 OpenClaw Skill 格式
